# Winter-fiber

```sh


if [ ! -d ~/Winter-fiber ]; then cd ~/ && git clone https://github.com/AndyInAi/Winter-fiber.git; fi


# 运行

cd ~/Winter-fiber && chmod +x ./Winter-fiber && ./Winter-fiber

# Starting HTTP server at http://0.0.0.0:8080/


# 测试

curl http://localhost:8080/


# 或者编译源代码后运行

# 编译源代码

cd ~/Winter-fiber && go build . 


# 运行

cd ~/Winter-fiber && ./Winter-fiber


# Starting HTTP server at http://0.0.0.0:8080/


# 测试
curl http://localhost:8080/


```
